   What are the equations for the lines of reflectional symmetry that exist for the square?
      \begin{prompt}
    \begin{multipleChoice}
      \choice{$R_{90}$, $F_{x=y}$, $F_{x=-y}$, and $F_{y=0}$.}
      \choice{$R_{45}$, $R_{135}$, $F_{x=-y}$, and $F_{x=y}$.}
      \choice{$F_{y=0}$, $F_{x=1}$, $F_{x=-y}$, and $F_{x=y}$.}
      \choice[correct]{$F_{y=0}$, $F_{x=0}$, $F_{x=-y}$, and $F_{x=y}$.}
    \end{multipleChoice}
  \end{prompt}

\end{exercise}
