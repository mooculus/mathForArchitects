\begin{exercise}
   Label the vertices on a square with four different labels. How many consecutive rotations are needed to return the vertices of the square to their original position?
      \begin{prompt}
    \begin{multipleChoice}
      \choice{ If we are rotating 180 degrees, then two times. If we are rotating 90 degrees, then four times. Finally, if we are rotating 270 degrees, then six times.}
      \choice[correct]{If we are rotating 90 degrees, then four times. If we are rotating 180 degrees, then two times. Finally, if we are rotating 270 degrees, then four times.}
    \end{multipleChoice}
  \end{prompt}

\end{exercise}
