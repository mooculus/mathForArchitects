\begin{exercise}
  Letting $\mat{F} = \mat{F}_{\ell}$ and $\mat{R}= \mat{R}_{120}$, which of
  the following symmetries of the equilateral triangle are equal to
  $\mat{R}_{120}$?
  \begin{selectAll}
    \choice{$\mat{F}\mat{R}\circ\mat{F}$}
    \choice{$\mat{F}\mat{R}^2\circ\mat{F}\mat{R}$}
    \choice{$\mat{R}\circ \mat{R}$}
    \choice[correct]{$\mat{F}\mat{R}\circ \mat{F}\mat{R}^2$}
    \choice[correct]{$\mat{F}\circ \mat{F}\mat{R}$}
    \choice{$\mat{F}\circ\mat{F}\mat{R}^2$}
    \choice[correct]{$\mat{F}\mat{R}^2\circ \mat{F}$}
    \choice[correct]{$\mat{R}^2\circ \mat{R}^2$}

  \end{selectAll}
\end{exercise}
