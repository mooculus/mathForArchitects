%% N-GON code
\tikzset{
    pics/tikzngon/.style n args={2}{
        code={
        \tikzmath{\n = #1;\s = #2;\rr=6;\m=\n-1;\dd=.4;}
        \foreach \i in {0,1,...,\m}
          {
            \draw[ultra thick,rounded corners=.05mm]
            ({\rr*sin(\i *360/\n)},{\rr*cos(\i *360/\n)}) --
            ({\rr*sin((\i+\s)*360/\n)},{\rr*cos((\i+\s)*360/\n)});
          }
          \foreach \i in {0,1,...,\m}
            {\pgfmathtruncatemacro{\xy}{\i+1}
             \node[circle,fill=black,inner sep=0pt, minimum size=13pt,text=white]
             at ({(\rr)*sin((mod(\i*\s,\n))*360/(\n))},{(\rr)*cos((mod(\i*\s,\n))*360/\n)}) {\normalfont\bfseries\sffamily{\xy}};
           }
        }}}
