\begin{sagesilent}
  a = randint(-5,5)
  b = randint(-5,5)
\end{sagesilent}

\begin{exercise}
  Considering the point $(\sage{a*sqrt(2)},\sage{b*sqrt(2)})$, use a
  matrix to rotate this point $45^\circ$ about the origin.
  \begin{prompt}
    \[
    \mat{R}_{45}\vec{p} = \left(\answer{\sage{a-b}},\answer{\sage{a+b}}\right)
    \]
  \end{prompt}
\end{exercise}
