\makerandom
 
\begin{sagesilent}
  a = randint(-10,10)
  b = randint(-10,10)
  p = randint(-10,10)
  q = randint(-10,10)
\end{sagesilent}

\begin{exercise}
  Compute the distance between the points $\vec{p} =
  (\sage{a},\sage{b})$ and $\vec{q} = (\sage{p},\sage{q})$.
  \begin{prompt}
    \[
    d(\vec{p},\vec{p}) = \answer{\sage{sqrt((a-p)^2 + (b-q)^2)}}
    \]
  \end{prompt}
\end{exercise}
