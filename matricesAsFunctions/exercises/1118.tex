\begin{exercise}
  The matrix $\mat{T}_{(\sage{u},\sage{v})}$ was used to translate the point
  $\vec{p}$ to $(\sage{a},\sage{b})$.  What is $\vec{p}$?
  \begin{prompt}
    \[
    \vec{p} = (\sage{a-u},\sage{b-v})
    \]
  \end{prompt}
\end{exercise}
