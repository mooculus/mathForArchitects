  \begin{prompt}
    \[
    \mat{F}=\begin{bmatrix}
      \answer{\sage{(1-m^2)/(1+m^2)}} & \answer{\sage{(2*m)/(1+m^2)}} & \answer{\sage{0}}\\
      \answer{\sage{(2*m)/(1+m^2)}} & \answer{\sage{(m^2-1)/(1+m^2)}} & \answer{\sage{0}}\\
      \answer{\sage{0}} & \answer{\sage{0}} & \answer{\sage{1}}
    \end{bmatrix}
    \]
  \end{prompt}
