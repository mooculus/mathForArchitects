  \begin{prompt}
    \[
    \mat{R}=\begin{bmatrix}
      \answer{\sage{cos(theta)}} & \answer{\sage{-sin(theta)}} & \answer{\sage{0}}\\
      \answer{\sage{sin(theta)}} & \answer{\sage{cos(theta)}} & \answer{\sage{0}}\\
      \answer{\sage{0}} & \answer{\sage{0}} & \answer{\sage{1}}
    \end{bmatrix}
    \]
  \end{prompt}
\end{exercise}
