\begin{document}
\begin{exercise}
  Select true statements:
  \begin{selectAll}
      \choice{There are exactly $5$ regular tessellations.}
      \choice[correct]{Any quadrilateral tessellates the plane.}
      \choice[correct]{Any triangle tessellates the plane.}
      \choice[correct]{If a triangle is used to tessellate the plane,
        there it is always the case that exactly $6$ angles will fit
        around each vertex.}
      \choice{If a polygon has more than $6$ sides, then it cannot tessellate
        the plane.}
  \end{selectAll}
\end{exercise}
