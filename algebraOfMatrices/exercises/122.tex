\begin{exercise}
  Consider the two translations $\mat{T}_{(\sage{a},\sage{b})}$ and $ \mat{T}_{(\sage{c},\sage{d})}$. Do these commute?

    \begin{prompt}
    \begin{multipleChoice}
      \choice[correct]{Yes, $\mat{T}_{(\sage{a},\sage{b})} \mat{T}_{(\sage{c},\sage{d})} = \mat{T}_{(\sage{c},\sage{d})} \mat{T}_{(\sage{a},\sage{b})}$.}
      \choice{No, $\mat{T}_{(\sage{a},\sage{b})} \mat{T}_{(\sage{c},\sage{d})} \neq \mat{T}_{(\sage{c},\sage{d})} \mat{T}_{(\sage{a},\sage{b})}$.}
    \end{multipleChoice}
  \end{prompt}
\end{exercise}
