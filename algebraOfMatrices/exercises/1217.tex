\begin{exercise}
  Given the point $\vec{p}=(\sage{a}, \sage{b})$, use matrices to compute $\vec{q}=\mat{T}_{(-6, -3)}\mat{R}_{135} \vec{p}$.


  \begin{prompt}
    \[
      \vec{q} = \left(\answer{\sage{ -6 - (1/sqrt(2)) * (a+b)}}, \answer{\sage{-3 + (1/sqrt(2)) * (a-b) }}\right)
    \]
  \end{prompt}
\end{exercise}
