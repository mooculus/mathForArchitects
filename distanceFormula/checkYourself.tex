\end{enumerate}
\end{exercise}

\begin{exercise}
  Consider this drawing of an ellipse:
  \begin{center}
    \begin{tikzpicture}[geometryDiagrams,scale=.8]
      \draw (0,0) ellipse (2in and 1in);
      \draw[->,thin] (0in,-1.5in) -- (0in,1.5in);
      \draw[->,thin] (-2.5in,0in) -- (2.5in,0in);
      \filldraw[black] (2in,0) circle (2pt) node[above right] {(4,0)};
      \filldraw[black] (0,1in) circle (2pt) node[above right] {(0,2)};
      
      \filldraw[black] (1.73in,0in) circle (2pt) node[below left] {(3,0)};
      \filldraw[black] (-1.73in,0in) circle (2pt) node[below right] {(-3,0)};
      
      \draw[dashed,thin] (-1.73in,0in) -- (.9in,.89in);
      \draw[dashed,thin] (1.73in,0in) -- (.9in,.89in);
      
      \node at (-.75in,.5in) {$X$};
      \node at (1.2in,.35in) {$Y$};
      
      \filldraw[black] (.9in,.89in) circle (2pt);
    \end{tikzpicture}
      \end{center}
The sum of the lengths of dotted lines $X+Y$ is:
\begin{enumerate}
\begin{multicols}{2}
 \item $6$
 \item $8$
 \item $10$
 \item $12$
\end{multicols}
\end{enumerate}
\end{exercise}
