\newcommand{\stackedcircles}[3][0pt]{%
  % #1: gap (optional), #2: count, #3: diameter
  \pgfmathtruncatemacro{\N}{#2-1}
  \pgfmathsetlengthmacro{\D}{#3}
  \pgfmathsetlengthmacro{\R}{0.5*\D}
  \pgfmathsetlengthmacro{\STEP}{\D + (#1)}
  \begin{tikzpicture}[line cap=round]
    \foreach \k in {0,...,\N} {
      \draw (0,\k*\STEP) circle (\R);
    }
  \end{tikzpicture}%
}

\begin{question}
  We obtain significant digits by acknowledging our imprecision in  measurement. 
  \begin{enumerate}
    \item Look up measurement rules for significant digits 
  online and \textbf{describe them here in your own words}.
   \item Use the rules of significant digits to measure this LEGO minifigure with each ruler below. Explain in words what you do as you solve the problem.
  \end{enumerate}
\begin{center}
  \begin{tikzpicture}[x=1.6cm,y=1.6cm,line cap=round]
    \def\RulerH{4}        % height in "units"
    \def\minor{0.15}       % minor tick length (cm)
    \def\major{0.35}       % major tick length (cm)
  
    % spine
    \draw[line width=0.6pt] (0,0) -- (0,\RulerH);
  
    % minor ticks every 0.1
    \foreach \y in {0,0.1,...,4} {
      \draw (0,\y) -- (\minor,\y);
    }
  
    % major ticks every 1.0
    \foreach \y in {0,1,...,4} {
      \draw[line width=0.8pt] (0,\y) -- (\major,\y);
    }
  
    % labels 1.0 .. 10.0
